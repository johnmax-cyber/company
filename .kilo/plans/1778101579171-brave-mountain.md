# Plan: Gradual Design Integration for Faith & Fashion Nairobi E-commerce

## Overview
Integrate specific design elements from the provided HTML templates into the existing e-commerce platform. The goal is to modernize the visual design while maintaining all existing functionality.

## Implementation Approach
Since I can only update JavaScript (per user request) and need to maintain backward compatibility, the approach is:

1. **JavaScript Updates Only** - Update selectors to work with existing DOM
2. **CSS Additions** - Add new styling via JavaScript-injected styles
3. **Progressive Enhancement** - Enhance existing elements without breaking them
4. **Component Updates** - Update JavaScript components to add modern interactions

## Design Elements to Integrate via JavaScript

### Key Upgrades
1. **Dynamic Styling Injection**
   - Inject modern CSS variables into :root

   - Apply gradient backgrounds dynamically

2. **Enhanced Interactions**

const designTokens = {
  '--primary': '#022448',
  '--secondary': '#b52426',
  '--background': '#f9f9ff',
  '--surface': '#ffffff',
  '--on-surface': '#111c2c',
  '--surface-variant': '#d8e3fa',
  '--outline': '#74777f',
  '--shadow': '0 20px 40px rgba(30,58,95,0.08)',
  '--shadow-lg': '0 30px 60px rgba(30,58,95,0.12)',
  '--radius': '1rem',
  '--radius-lg': '2rem',
  '--glass': 'rgba(255, 255, 255, 0.7)',
  '--backdrop-blur': 'blur(12px)'
};

// Inject design tokens
function injectDesignTokens() {
  const root = document.documentElement;
  Object.entries(designTokens).forEach(([key, value]) => {
    root.style.setProperty(key, value);
  });
}

// Add glass effect classes
function addGlassEffect(element) {
  if (element) {
    element.style.backdropFilter = 'blur(12px)';
    element.style.background = 'rgba(255, 255, 255, 0.7)';
    element.style.border = '1px solid rgba(255, 255, 255, 0.3)';
  }
}
```

// Toast notification system
class Toast {
  static show(message, type = 'info') {
    const existing = document.querySelector('.toast-notification');
    if (existing) existing.remove();
    
    const toast = document.createElement('div');
    toast.className = `toast-notification ${type}`;
    toast.innerHTML = `
      <span class="material-symbols-outlined">${this.getIcon(type)}</span>
      <span>${message}</span>
    `;
    
    document.body.appendChild(toast);
    
    setTimeout(() => toast.classList.add('show'), 10);
    setTimeout(() => {
      toast.classList.remove('show');
      setTimeout(() => toast.remove(), 300);
    }, 3000);
  }
  
  static getIcon(type) {
    return {
      success: 'check_circle',
      error: 'error',
      info: 'info',
      warning: 'warning'
    }[type] || 'info';
  }
}

// Badge creator
function createBadge(text, type = 'primary') {
  const badge = document.createElement('span');
  badge.className = `badge badge-${type}`;
  badge.textContent = text;
  return badge;
}

// Button style enhancer
function enhanceButton(button, style = 'primary') {
  const styles = {
    primary: 'bg-primary text-white hover:bg-primary-dark',
    secondary: 'bg-secondary text-white hover:bg-secondary-dark',
    outline: 'border border-primary text-primary hover:bg-primary hover:text-white',
    ghost: 'text-primary hover:bg-primary/10'
  };
  
  button.className = `btn ${styles[style]}`;
}
```

#### Shop Page Enhancements

// Enhanced shop page initialization
class EnhancedShopPage extends ShopPage {
  async init() {
    await super.init();
    this.enhanceShopUI();
  }
  
  enhanceShopUI() {
    // Add glass effect to filters
    const filterSection = document.querySelector('.filters-wrapper');
    if (filterSection) {
      addGlassEffect(filterSection.parentElement);
    }
    
    // Enhance product grid
    this.enhanceProductGrid();
  }
  
  enhanceProductGrid() {
    const grid = document.getElementById('productsGrid');
    if (!grid) return;
    
    // Override default rendering with enhanced version
    const originalRender = this.renderProducts;
    this.renderProducts = (products) => {
      originalRender.call(this, products);
      this.addEnhancementsToGrid();
    };
  }
  
  addEnhancementsToGrid() {
    // Add hover effects and quick add buttons
    document.querySelectorAll('.product-card').forEach(card => {
      card.style.transition = 'all 0.3s ease';
      card.addEventListener('mouseenter', (e) => {
        e.currentTarget.style.transform = 'translateY(-8px)';
        e.currentTarget.style.boxShadow = 'var(--shadow-lg)';
      });
      card.addEventListener('mouseleave', (e) => {
        e.currentTarget.style.transform = 'translateY(0)';
        e.currentTarget.style.boxShadow = 'var(--shadow)';
      });
    });
  }
}

// Enhanced product page
class EnhancedProductPage extends ProductPage {
  async init() {
    await super.init();
    this.enhanceProductUI();
  }
  
  enhanceProductUI() {
    // Add quick view functionality
    this.addQuickView();
    
    // Enhance related products
    this.enhanceRelatedProducts();

  }
  
  addQuickView() {
    const mainImage = document.querySelector('.product-main-image img');
    if (mainImage) {
      mainImage.style.cursor = 'zoom-in';
      mainImage.addEventListener('click', () => {
        this.showQuickView();
      });
    }
  }
  
  showQuickView() {
    // Create quick view modal
    const modal = document.createElement('div');
    modal.className = 'quick-view-modal';
    modal.innerHTML = `
      <div class="quick-view-content glass-effect">
        <button class="close-modal">&times;</button>
        <img src="${this.product.image_url || ''}" alt="${this.product.name}" />
        <h2>${this.product.name}</h2>
        <p class="price">KSh ${this.product.price.toLocaleString()}</p>
        <button class="btn btn-primary btn-lg" id="quickAddBtn">Add to Cart</button>
      </div>
    `;
    
    document.body.appendChild(modal);
    
    modal.querySelector('.close-modal').addEventListener('click', () => modal.remove());
    modal.querySelector('#quickAddBtn').addEventListener('click', () => {
      cartService.addItem({
        id: this.product.id,
        name: this.product.name,
        price: this.product.price,
        qty: 1
      });
      Toast.show(`${this.product.name} added to cart`, 'success');
      modal.remove();
      this.updateCartCount();
    });
  }
  
  enhanceRelatedProducts() {
    const relatedSection = document.getElementById('relatedSection');
    if (relatedSection) {
      relatedSection.style.marginTop = '4rem';
      relatedSection.querySelector('h2').style.fontFamily = 'var(--font-family-heading)';
    }
  }
  
  addQuantitySelector() {
    const container = document.getElementById('productDetail');
    if (!container) return;
    
    container.addEventListener('click', (e) => {
      if (e.target.classList.contains('qty-btn')) {
        const qtySpan = e.target.parentElement.querySelector('.qty');
        let qty = parseInt(qtySpan.textContent);
        
        if (e.target.classList.contains('plus')) {
          qty++;
        } else if (e.target.classList.contains('minus') && qty > 1) {
          qty--;
        }
        
        qtySpan.textContent = qty;
      }
    });
  }
}

import { HomePage } from './pages/home.js';
import { ShopPage } from './pages/shop.js';
import { CartPage } from './pages/cart.js';
import { ProductPage } from './pages/product.js';

// Initialize design tokens first
function initializeDesignSystem() {
  // Inject modern design tokens
  const root = document.documentElement;
  root.style.setProperty('--primary', '#022448');
  root.style.setProperty('--secondary', '#b52426');
  root.style.setProperty('--background', '#f9f9ff');

  root.style.setProperty('--on-surface', '#111c2c');

  root.style.setProperty('--radius', '1rem');
  root.style.setProperty('--radius-lg', '2rem');
}

// Enhanced application initialization
document.addEventListener('DOMContentLoaded', async () => {
  try {
    // Initialize design system
    initializeDesignSystem();
    
    const path = window.location.pathname;
    let page;
    
    // Detect enhanced or standard page based on path
    const useEnhanced = true; // Toggle this to switch between enhanced/standard
    
    if (path.includes('/shop')) {
      page = useEnhanced ? new EnhancedShopPage() : new ShopPage();
    } else if (path.includes('/cart')) {
      page = new CartPage(); // Use enhanced cart
    } else if (path.includes('/product')) {
      page = useEnhanced ? new EnhancedProductPage() : new ProductPage();
    } else {
      page = new HomePage();
    }
    
    await page.init();
    
    // Add global enhancements
    addGlobalEnhancements();
  } catch (error) {
    console.error('Failed to initialize application:', error);
  }
});

// Global UI enhancements
function addGlobalEnhancements() {
  // Add smooth scroll to all anchor links
  document.querySelectorAll('a[href^="#"]').forEach(anchor => {
    anchor.addEventListener('click', function (e) {
      e.preventDefault();
      const target = document.querySelector(this.getAttribute('href'));
      if (target) {
        target.scrollIntoView({ behavior: 'smooth' });
      }
    });
  });
  
  // Add glass effect to headers where appropriate
  const header = document.querySelector('.header');
  if (header && window.scrollY > 50) {
    addGlassEffect(header);
  }
  
  window.addEventListener('scroll', () => {
    if (header) {
      if (window.scrollY > 50) {
        addGlassEffect(header);
      } else {
        header.style.background = 'var(--surface)';
        header.style.backdropFilter = 'none';
      }
    }
  });
}
